bouncingtext.py
Creates bouncing colorful text that moves across your screen with the message "YOU ARE AN IDIOT!"
from win32api import *
from win32gui import *
import time
import random

def main():
    x, y = 0, 0
    signX, signY = 1, 1
    incrementor = 5
    while True:
        w, h, hdc = GetSystemMetrics(0), GetSystemMetrics(1), GetDC(0)
        text = "YOU ARE AN IDIOT!"
        color = random.randint(0, 255) | (random.randint(0, 255) << 8) | (random.randint(0, 255) << 16)
        SetTextColor(hdc, color)
        SetBkMode(hdc, 0)
        ExtTextOut(hdc, x, y, 0, None, text, None)
        x += incrementor * signX
        y += incrementor * signY
        if y >= h or y <= 0:
            signY *= -1
        if x >= w or x <= 0:
            signX *= -1
        time.sleep(0.01)
        ReleaseDC(0, hdc)

main()
text.py
Displays random colorful "You Are An Idiot" text in various positions across your screen.
from win32api import *
from win32gui import *
import time
import random

def main():
    while True:
        w, hdc = GetSystemMetrics(0), GetDC(0)
        text = "You Are An Idiot"
        color = random.randint(0, 255) | (random.randint(0, 255) << 8) | (random.randint(0, 255) << 16)
        SetTextColor(hdc, color)
        SetBkMode(hdc, 0)
        x, y = random.randint(0, w), random.randint(0, w)
        ExtTextOut(hdc, x, y, 0, None, text, None)
        time.sleep(0.001)
        ReleaseDC(0, hdc)

main()
massduplicate.py
Creates a visual effect by duplicating and randomly placing screen sections, creating a fragmented appearance.
from win32api import *
from win32con import *
from win32gui import *
import random

def main():
    while True:
        w, h, hdc = GetSystemMetrics(0), GetSystemMetrics(1), GetDC(0)
        BitBlt(hdc, random.randint(-500, 500), random.randint(-500, 500), w, h, hdc, random.randint(-500, 500), random.randint(-500, 500), SRCCOPY)
        ReleaseDC(0, hdc)

main()
flash.py
Creates a flashing effect by inverting the colors on your screen at regular intervals. Warning: May be uncomfortable for those with photosensitivity.
# thanks to malwux on discord for giving me this code (had alot of syntax errors but i fixed and modified it)

from win32api import *
from win32con import *
from win32gui import *
import time

def main():
    while True:
        w, h, hdc = GetSystemMetrics(0), GetSystemMetrics(1), GetDC(0)
        PatBlt(hdc, 0, 0, w, h, PATINVERT)
        time.sleep(1)
        ReleaseDC(0, hdc)

main()
